SELECT * FROM FUNCIONARIO


SELECT NOMEDEPT, COUNT(NOMEFUNC) FROM FUNCIONARIO GROUP BY
NOMEDEPT ;
SELECT NOMEDEPT, COUNT(*) FROM FUNCIONARIO WHERE COMISSAO >0
GROUP BY NOMEDEPT;
SELECT NOMEDEPT, COUNT(NOMEDEPT) FROM FUNCIONARIO WHERE
extract(YEAR FROM DTADIMISSAO) = 2000 GROUP BY NOMEDEPT;
SELECT CARGO , COUNT(*) FROM FUNCIONARIO GROUP BY CARGO;
SELECT CARGO, Count (*) As "DEPARTAMENTOS COM MAIS DE DOIS
FUNCIONARIOS" From FUNCIONARIO Group By Cargo Having Count (*) >2;
SELECT NOMEDEPT, Count (*) As "DEPARTAMENTOS COM MAIS DE UM
FUNCIONARIO" From FUNCIONARIO Group By NOMEDEPT Having Count (*) >1;
SELECT CARGO, Max(SALARIO) AS "Maior Salário" FROM FUNCIONARIO GROUP BY
CARGO;
ORDER BY SALARIO ASC;
SELECT Cargo, Min(Salario) AS "Menor Salário" FROM FUNCIONARIO GROUP BY
CARGO;
ORDER BY SALARIO ASC;
SELECT CARGO, floor(Avg(SALARIO)) AS "Media Salário" FROM FUNCIONARIO
GROUP BY CARGO;
ORDER BY SALARIO ASC;
SELECT CARGO, Avg (SALARIO) AS "Media Salario" FROM FUNCIONARIO GROUP
BY CARGO Order By avg(SALARIO) Desc;
SELECT NOMEDEPT, Avg (SALARIO) AS "Media Salario" FROM FUNCIONARIO
GROUP BY NOMEDEPT Order By Avg(SALARIO);
SELECT NOMEDEPT, Avg (SALARIO) AS "Media Salario" FROM FUNCIONARIO
Where SALARIO >3000 GROUP BY NOMEDEPT Order By Avg(SALARIO) ;
SELECT NOMEDEPT, Avg (SALARIO) AS "Media Salario" FROM FUNCIONARIO
Where SALARIO <=3000 GROUP BY NOMEDEPT Order By Avg(SALARIO) ;
SELECT NOMEDEPT,CARGO, COUNT(IDFUNC) AS "Numero Funcionario" FROM
FUNCIONARIO GROUP BY NOMEDEPT,CARGO;
SELECT CARGO,NOMEDEPT, COUNT(IDFUNC) AS "Numero Funcionario" FROM
FUNCIONARIO GROUP BY CARGO,NOMEDEPT;
SELECT CARGO, Extract(YEAR FROM DTADIMISSAO) As Ano_Admissao,
Avg(SALARIO) * 12 As Media_Salario_Anual From FUNCIONARIO Group By CARGO,
Extract(Year From DTADIMISSAO) Order By CARGO, Media_Salario_Anual;
